## Deployment

### Podman Compose

```yaml
services:
  redis:
    image: ghcr.io/daemonless/redis:latest
    container_name: redis
    environment:
      - PUID=1000
      - PGID=1000
      - TZ=UTC
    volumes:
      - "/path/to/containers/redis:/config"
    ports:
      - 6379:6379
    restart: unless-stopped
```
